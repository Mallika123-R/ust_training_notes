package javaeight;

import java.util.function.BiFunction;
import java.util.function.Function;

public class FunctionalLambda {

    public static void main(String[] args) {
        // predefined functional interface
                 // input ,output
        Function<Integer,Integer>  result = (Integer input) ->  {
            int output =input+100;
            return output;
        };
        // apply method ;
        int ans =result.apply(100);
        System.out.println(ans);
        Function<Integer,String> ageFunc = (age) -> {
            return "Your Age is " + age;
        };
        String message=ageFunc.apply(18);
        System.out.println(message);

        BiFunction<Double,Double,String > areaRect=(length,bredth) ->{
            double area=length*bredth;
            return  "Area " + area;
        };
        System.out.println(areaRect.apply(12.34,4.5));
    }

}

package employeestreams;

import java.sql.SQLOutput;
import java.util.Comparator;
import java.util.List;
import java.util.stream.Collectors;

public class EmployeeMain {
    static {
        new EmployeeDao().addEmployee(new Employee(1,"A",25000));
        new EmployeeDao().addEmployee(new Employee(3,"H",45000));
        new EmployeeDao().addEmployee(new Employee(2,"G",65000));
        new EmployeeDao().addEmployee(new Employee(5,"P",75000));
        new EmployeeDao().addEmployee(new Employee(4,"Z",85000));
    }

    public static void main(String[] args) {
        List<Employee> employeeList = new EmployeeDao().getEmployeeList();
        // List Employeess - salary -> 50000

        List<Employee> empSalarygreaterthan50 = employeeList.stream()
                .filter((employee -> employee.getSalary() > 5000))
                .collect(Collectors.toList());
        empSalarygreaterthan50.forEach(employee -> {
            System.out.println(employee.getId() + "---" + employee.getName()+"---"+ employee.getSalary());
        });

        // list of employeess - salary 50000 -1 lac

        System.out.println("increment salary by 20000 whose salary<50000");
        List<Employee> result2 = employeeList.stream()
                .filter(employee -> employee.getSalary() < 50000)
                .map(employee -> {
                    employee.setSalary(employee.getSalary() + 20000);
                    return employee;
                })
                .collect(Collectors.toList());
        result2.forEach(System.out::println);

        System.out.println("Asecneding order by employee name");
        employeeList.stream()
                .sorted(Comparator.comparing(Employee::getName).reversed()) // Employee::getName -->employee -> employee.getName()
                .forEach(System.out::println);
        System.out.println("Ascending Order by employee id");
        employeeList.stream()
                .sorted(Comparator.comparing(Employee::getId))
                .forEach(System.out::println);
        System.out.println("Print Only Employee Names slaray> 50000");
        List<String> names=employeeList.stream()
                .map(employee -> employee.getName().toUpperCase())
                .sorted()
                .collect(Collectors.toList());
        System.out.println(names);
    }

}
