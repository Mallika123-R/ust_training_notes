package com.kewords;

public class ThisKeywordEx {
    // Instance variable
    int length=555;
    int bredth=666;
    public ThisKeywordEx(int length, int bredth) {
        this.length = length;
        this.bredth = bredth;
    }
    public ThisKeywordEx() {

    }
    void display(){
        // local variables
        int length=100;
        int bredth=200;

        System.out.println(length);       // 100
        System.out.println(this.length);  // 555

        System.out.println(bredth);       // 200
        System.out.println(this.bredth);  // 666
    }
    public static void main(String[] args) {
        new ThisKeywordEx().display();
    }

}

package com.kewords;

class A {
    int a=100;
    A() {
        super();  // first line of constructor
        System.out.println("A");
        System.out.println(a);
    }
}

class B extends A {
    int a=200;
     B() {
        super();
        System.out.println("B");
         System.out.println(super.a);
         System.out.println(a);
    }
}

public class C extends B {
    int a=300;
    C() {
        super();
        System.out.println("C");
        System.out.println(super.a);
        System.out.println(a);
    }

    public static void main(String[] args) {
        C c = new C();     //  A B C

        B  b=new B();    // B c

        A a=new A();     // A
    }
}
